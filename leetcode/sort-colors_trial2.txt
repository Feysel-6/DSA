class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        counts = [0] * (max(nums) + 1)
        for num in nums:
            counts[num] += 1
        target = 0
        for index,value in enumerate(counts):
            for j in range(value):
                nums[target] = index
                target += 1
